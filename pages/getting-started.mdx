# Getting Started

<Callout type="info">
  This page uses JavaScript and npm. JavaScript is not necessary to use
  Interweave. See the [build page](/build-do-it-yourself) for more information.
</Callout>

## Installation

<Steps>
### Step 1: Install
```bash
npm install @interweave/interweave
```

### Step 2: Build Config

<Tabs items={["JavaScript", "TypeScript"]}>
    <Tab>
        ```ts filename="interweave.js"
        import { buildInterface } from "@interweave/interweave";

        // Define the content of your interface
        const configuration = {
          keys: {
            firstName: {
              schema: {
                type: "string"
              }
            }
          },
          requests: {
            create: {
              uri: "https://example.com/names",
              method: "POST"
            }
          }
        };

        // Send to Interweave
        buildInterface("unique-key", configuration, projectId);

````
    </Tab>
    <Tab>
        ```ts filename="interweave.ts"
        import { type Schema, buildInterface } from "@interweave/interweave";

        // Define the content of your interface
        const configuration: Schema = {
          keys: {
            firstName: {
              schema: {
                type: "string"
              }
            }
          },
          requests: {
            create: {
              uri: "https://example.com/names",
              method: "POST"
            }
          }
        };

        // Send to Interweave
        buildInterface("unique-key", configuration, projectId);

````

    </Tab>

</Tabs>
